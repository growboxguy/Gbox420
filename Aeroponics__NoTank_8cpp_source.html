74 void Aeroponics_NoTank::updateState(AeroNoTankStates NewState)
75 {
76  bool BlockOverWritingState = false; // Used when a state transitions to a new state
77  if (State != NewState && Debug)
78  {
79  memset(&LongMessage[0], 0, MaxLongTextLength);
80  strcat_P(LongMessage, (PGM_P)Name);
81  strcat_P(LongMessage, (PGM_P)F(" state: "));
82  strcat(LongMessage, toText_aeroNoTankState(State));
83  strcat_P(LongMessage, (PGM_P)F(" -> "));
84  strcat(LongMessage, toText_aeroNoTankState(NewState));
85  logToSerials(&LongMessage, true, 3);
86  }
87 
88  switch (NewState)
89  {
90  case AeroNoTankStates::DISABLED:
91  if (State != NewState)
92  {
93  Pump->stopPump();
94  }
95  break;
96  case AeroNoTankStates::IDLE:
97  if (State != NewState)
98  {
99  Pump->stopPump();
100  }
101  uint32_t Interval;
102  if (DayMode)
103  {
104  Interval = DayInterval * 60000;
105  }
106  else
107  {
108  Interval = NightInterval * 60000;
109  }
110  if (millis() - SprayTimer >= Interval)
111  {
112  updateState(AeroNoTankStates::PRIMING);
113  BlockOverWritingState = true;
114  }
115  break;
116  case AeroNoTankStates::PRIMING:
117  if (State != NewState)
118  {
119  Pump->startPump(true);
120  }
121  if (Pump->getState() != PressurePumpStates::PRIMING)
122  {
123  updateState(AeroNoTankStates::SPRAY);
124  BlockOverWritingState = true;
125  }
126  break;
127  case AeroNoTankStates::SPRAY:
128  if (State != NewState)
129  {
130  SprayTimer = millis();
131  }
132  if (FeedbackPressureSensor->readPressure(false) > MaxPressure || millis() - SprayTimer >= Duration * 1000)
133  {
134  LastSprayPressure = FeedbackPressureSensor->getPressure();
135  updateState(AeroNoTankStates::RELEASE);
136  BlockOverWritingState = true;
137  }
138  break;
139  case AeroNoTankStates::RELEASE: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
140  if (State != NewState)
141  {
142  Pump->stopPump();
143  }
144  if (Pump->getState() == PressurePumpStates::IDLE)
145  {
146  updateState(AeroNoTankStates::IDLE);
147  BlockOverWritingState = true;
148  }
149  if (Pump->getState() == PressurePumpStates::DISABLED)
150  {
151  updateState(AeroNoTankStates::DISABLED);
152  BlockOverWritingState = true;
153  }
154  break;
155  case AeroNoTankStates::MIX: //< Release remaining pressure from the misting loop AND pump output side
156  if (State != NewState)
157  {
158  Pump->startMixing();
159  }
160  if (Pump->getState() != PressurePumpStates::MIX)
161  {
162  updateState(AeroNoTankStates::IDLE);
163  BlockOverWritingState = true;
164  }
165  break;
166  }
167 
168  if (State != NewState && !BlockOverWritingState)
169  {
170  State = NewState;
171  }
172 }
