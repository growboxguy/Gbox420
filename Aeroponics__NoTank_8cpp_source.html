66 void Aeroponics_NoTank::updateState(AeroNoTankStates NewState)
67 {
68  bool BlockOverWritingState = false; //Used when a state transitions to a new state
69  if (State != NewState && *Debug)
70  {
71  memset(&LongMessage[0], 0, MaxLongTextLength);
72  strcat_P(LongMessage, (PGM_P)Name);
73  strcat_P(LongMessage, (PGM_P)F(" state: "));
74  strcat(LongMessage, toText_aeroNoTankState(State));
75  strcat_P(LongMessage, (PGM_P)F(" -> "));
76  strcat(LongMessage, toText_aeroNoTankState(NewState));
77  logToSerials(&LongMessage, true, 3);
78  }
79 
80  switch (NewState)
81  {
82  case AeroNoTankStates::DISABLED:
83  if (State != NewState)
84  {
85  Pump->stopPump();
86  }
87  break;
88  case AeroNoTankStates::IDLE:
89  if (State != NewState)
90  {
91  Pump->stopPump();
92  }
93  uint32_t Interval;
94  if (DayMode)
95  {
96  Interval = *DayInterval * 60000;
97  }
98  else
99  {
100  Interval = *NightInterval * 60000;
101  }
102  if (millis() - SprayTimer >= Interval)
103  {
104  updateState(AeroNoTankStates::PRIMING);
105  BlockOverWritingState = true;
106  }
107  break;
108  case AeroNoTankStates::PRIMING:
109  if (State != NewState)
110  {
111  Pump->startPump(true);
112  }
113  if (Pump->getState() != PressurePumpStates::PRIMING)
114  {
115  updateState(AeroNoTankStates::SPRAY);
116  BlockOverWritingState = true;
117  }
118  break;
119  case AeroNoTankStates::SPRAY:
120  if (State != NewState)
121  {
122  SprayTimer = millis();
123  }
124  if (FeedbackPressureSensor->readPressure(false) > *MaxPressure || millis() - SprayTimer >= *Duration * 1000)
125  {
126  LastSprayPressure = FeedbackPressureSensor->getPressure();
127  updateState(AeroNoTankStates::RELEASE);
128  BlockOverWritingState = true;
129  }
130  break;
131  case AeroNoTankStates::RELEASE: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
132  if (State != NewState)
133  {
134  Pump->stopPump();
135  }
136  if (Pump->getState() == PressurePumpStates::IDLE)
137  {
138  updateState(AeroNoTankStates::IDLE);
139  BlockOverWritingState = true;
140  }
141  if (Pump->getState() == PressurePumpStates::DISABLED)
142  {
143  updateState(AeroNoTankStates::DISABLED);
144  BlockOverWritingState = true;
145  }
146  break;
147  case AeroNoTankStates::MIX: //< Release remaining pressure from the misting loop AND pump output side
148  if (State != NewState)
149  {
150  Pump->startMixing();
151  }
152  if (Pump->getState() != PressurePumpStates::MIX)
153  {
154  updateState(AeroNoTankStates::IDLE);
155  BlockOverWritingState = true;
156  }
157  break;
158  }
159 
160  if (State != NewState && !BlockOverWritingState)
161  {
162  State = NewState;
163  }
164 }
