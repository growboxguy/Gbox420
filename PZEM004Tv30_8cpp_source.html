223bool PZEM004Tv30::sendCmd8(uint8_t cmd, uint16_t rAddr, uint16_t val, bool check, uint16_t slave_addr){
224 uint8_t sendBuffer[8]; // Send buffer
225 uint8_t respBuffer[8]; // Response buffer (only used when check is true)
226
227 if((slave_addr == 0xFFFF) ||
228 (slave_addr < 0x01) ||
229 (slave_addr > 0xF7)){
230 slave_addr = _addr;
231 }
232
233 sendBuffer[0] = slave_addr; // Set slave address
234 sendBuffer[1] = cmd; // Set command
235
236 sendBuffer[2] = (rAddr >> 8) & 0xFF; // Set high byte of register address
237 sendBuffer[3] = (rAddr) & 0xFF; // Set low byte =//=
238
239 sendBuffer[4] = (val >> 8) & 0xFF; // Set high byte of register value
240 sendBuffer[5] = (val) & 0xFF; // Set low byte =//=
241
242 setCRC(sendBuffer, 8); // Set CRC of frame
243
244 _serial->write(sendBuffer, 8); // send frame
245
246 if(check) {
247 if(!recieve(respBuffer, 8)){ // if check enabled, read the response
248 return false;
249 }
250
251 // Check if response is same as send
252 for(uint8_t i = 0; i < 8; i++){
253 if(sendBuffer[i] != respBuffer[i])
254 return false;
255 }
256 }
257 return true;
258}

363bool PZEM004Tv30::updateValues()
364{
365 //static uint8_t buffer[] = {0x00, CMD_RIR, 0x00, 0x00, 0x00, 0x0A, 0x00, 0x00};
366 static uint8_t response[25];
367
368 // If we read before the update time limit, do not update
369 if(_lastRead + UPDATE_TIME > millis()){
370 return true;
371 }
372
373 // Read 10 registers starting at 0x00 (no check)
374 sendCmd8(CMD_RIR, 0x00, 0x0A, false);
375
376
377 if(recieve(response, 25) != 25){ // Something went wrong
378 return false;
379 }
380
381
382
383 // Update the current values
384 _currentValues.voltage = ((uint32_t)response[3] << 8 | // Raw voltage in 0.1V
385 (uint32_t)response[4])/10.0;
386
387 _currentValues.current = ((uint32_t)response[5] << 8 | // Raw current in 0.001A
388 (uint32_t)response[6] |
389 (uint32_t)response[7] << 24 |
390 (uint32_t)response[8] << 16) / 1000.0;
391
392 _currentValues.power = ((uint32_t)response[9] << 8 | // Raw power in 0.1W
393 (uint32_t)response[10] |
394 (uint32_t)response[11] << 24 |
395 (uint32_t)response[12] << 16) / 10.0;
396
397 _currentValues.energy = ((uint32_t)response[13] << 8 | // Raw Energy in 1Wh
398 (uint32_t)response[14] |
399 (uint32_t)response[15] << 24 |
400 (uint32_t)response[16] << 16) / 1000.0;
401
402 _currentValues.frequeny =((uint32_t)response[17] << 8 | // Raw Frequency in 0.1Hz
403 (uint32_t)response[18]) / 10.0;
404
405 _currentValues.pf = ((uint32_t)response[19] << 8 | // Raw pf in 0.01
406 (uint32_t)response[20])/100.0;
407
408 _currentValues.alarms = ((uint32_t)response[21] << 8 | // Raw alarm value
409 (uint32_t)response[22]);
410
411 // Record current time as _lastRead
412 _lastRead = millis();
413
414 return true;
415}

452uint16_t PZEM004Tv30::recieve(uint8_t *resp, uint16_t len)
453{
454 #ifdef PZEM004_SOFTSERIAL
455 if(_isSoft)
456 ((SoftwareSerial *)_serial)->listen(); // Start software serial listen
457 #endif
458
459 unsigned long startTime = millis(); // Start time for Timeout
460 uint8_t index = 0; // Bytes we have read
461 while((index < len) && (millis() - startTime < READ_TIMEOUT))
462 {
463 if(_serial->available() > 0)
464 {
465 uint8_t c = (uint8_t)_serial->read();
466
467 resp[index++] = c;
468 }
469 yield(); // do background netw tasks while blocked for IO (prevents ESP watchdog trigger)
470 }
471
472 // Check CRC with the number of bytes read
473 if(!checkCRC(resp, index)){
474 return 0;
475 }
476
477 return index;
478}

490bool PZEM004Tv30::checkCRC(const uint8_t *buf, uint16_t len){
491 if(len <= 2) // Sanity check
492 return false;
493
494 uint16_t crc = CRC16(buf, len - 2); // Compute CRC of data
495 return ((uint16_t)buf[len-2] | (uint16_t)buf[len-1] << 8) == crc;
496}

509void PZEM004Tv30::setCRC(uint8_t *buf, uint16_t len){
510 if(len <= 2) // Sanity check
511 return;
512
513 uint16_t crc = CRC16(buf, len - 2); // CRC of data
514
515 // Write high and low byte to last two positions
516 buf[len - 2] = crc & 0xFF; // Low byte first
517 buf[len - 1] = (crc >> 8) & 0xFF; // High byte second
518}

569uint16_t PZEM004Tv30::CRC16(const uint8_t *data, uint16_t len)
570{
571 uint8_t nTemp; // CRC table index
572 uint16_t crc = 0xFFFF; // Default value
573
574 while (len--)
575 {
576 nTemp = *data++ ^ crc;
577 crc >>= 8;
578 crc ^= (uint16_t)pgm_read_word(&crcTable[nTemp]);
579 }
580 return crc;
581}
