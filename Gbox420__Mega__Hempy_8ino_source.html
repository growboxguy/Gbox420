264 time_t getNtpTime()
265 {
266  time_t NTPResponse = 0;
267  if (!SyncInProgress)
268  { // block calling the sync again inside an interrupt while already waiting for a sync to finish
269  SyncInProgress = true;
270  uint32_t LastRefresh = millis();
271  logToSerials(F("Waiting for NTP time"), false, 0);
272  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
273  {
274  NTPResponse = ESPCmd.GetTime();
275  delay(1000);
276  logToSerials(F("."), false, 0);
277  wdt_reset();
278  }
279  SyncInProgress = false;
280  if (NTPResponse == 0)
281  {
282  logToSerials(F("sync failed"), true, 3);
283  }
284  else
285  logToSerials(F("synchronized"), true, 3);
286  }
287  return NTPResponse;
288 }
