47 void HempyBucket::update_state(HempyStates NewState, bool Force)
48 {
49 if (UpdateInProgress && !Force)
50 {
51 return; // Exit if an update is already in progress
52 }
53 UpdateInProgress = true;
54 bool BlockOverWritingState = false; // Used when a state transitions to a new state
55 uint32_t CurrentTime = millis();
56 if (State != NewState)
57 {
58 StateTimer = CurrentTime; // Start measuring the time spent in the new State
59 ESP_LOGI("hempy", "%s State: %s -> %s", Name.c_str(), to_text_state(State), to_text_state(NewState));
60 }
61
62 switch (NewState)
63 {
64 case HempyStates::DISABLED:
65 if (WaterPump->state)
66 WaterPump->turn_off();
67 if (State != NewState) // When the state just changed
68 update_interval(DefaultUpdateInterval); // Restore the original update_interval from YAML
69 break;
70 case HempyStates::DRY:
71 if (WaterPump->state)
72 WaterPump->turn_off(); // Pump is considered failed. With automatic watering it should never switch to DRY mode. Triggers a notification.
73 if (State != NewState) // When the state just changed
74 update_interval(1000);
75 if (WeightSensor->state >= MaxWateringWeight->state || WeightSensor->state >= WetWeight->state) // Check if manual watering reached the Max Weight
76 {
77 update_state(HempyStates::IDLE, true);
78 BlockOverWritingState = true;
79 }
80 break;
81 case HempyStates::IDLE:
82 if (WaterPump->state)
83 WaterPump->turn_off();
84 if (State != NewState) // When the state just changed
85 update_interval(DefaultUpdateInterval); // Restore the original update_interval from YAML
86 if (!AverageReset && ((StartWateringWeight->state > 0 && AverageWeight <= StartWateringWeight->state) || (EvaporationTargetWeight->state > 0 && DryWeight->state > 0 && AverageWeight <= DryWeight->state)))
87 {
88 update_state(HempyStates::WATERING, true);
89 BlockOverWritingState = true;
90 }
91 break;
92 case HempyStates::WATERING:
93 if (State != NewState)
94 {
95 update_interval(1000); // 1sec - Speed up calling update()
96 StateWeight = WeightSensor->state; // Store the bucket weight before starting the pump
97 if (State != HempyStates::DRAINING) // First time entering the WATERING-DRAINING cycles
98 {
99 PumpOnTimer = CurrentTime;
100 WateringTimer = 0;
101 }
102 if (State == HempyStates::DRAINING)
103 {
104 PumpOnTimer = CurrentTime;
105 }
106 WaterPump->turn_on();
107 }
108 if (WeightSensor->state >= MaxWateringWeight->state || (WeightSensor->state >= StateWeight + WateringIncrement->state && WeightSensor->state >= WetWeight->state)) // Max weight reached OR (WateringIncrement's worth of water was added AND WetWeight is reached), wait for it to drain
109 {
110 WateringTimer += CurrentTime - PumpOnTimer;
111 update_state(HempyStates::DRAINING, true);
112 BlockOverWritingState = true;
113 }
114 else if ((CurrentTime - PumpOnTimer) > MaxWateringTime->state * 1000) // Disable watering in case MaxWateringTime is reached: Consider the pump broken
115 {
116 ESP_LOGW("Hempy", "%s Timeout, pump failed", Name.c_str());
117 update_state(HempyStates::DRY, true); // Switch to DRY state - Triggers alert in Home Assistant
118 BlockOverWritingState = true;
119 }
120 break;
121 case HempyStates::DRAINING:
122 if (WaterPump->state) // If the pump is on -> turn it off
123 WaterPump->turn_off();
124 if (State != NewState)
125 {
126 StateWeight = WeightSensor->state;
127 ESP_LOGW("Hempy", "Stored drain start weight: %.2f", StateWeight);
128 }
129 if (CurrentTime - StateTimer >= (DrainWaitTime->state * 1000)) // Waiting for the water to drain
130 {
131 DrainProgress += StateWeight - WeightSensor->state; // Calculate how much water has drained
132 // ESP_LOGW("Hempy", "if %.2f <= %.2f - %.2f ", WeightSensor->state, StateWeight, DrainTargetWeight->state);
133 if (DrainProgress >= DrainTargetWeight->state || WeightSensor->state >= MaxWateringWeight->state) // Check if enough water was drained into the waste reservoir, OR MaxWateringWeight is reached
134 {
135 WetWeight->publish_state(WeightSensor->state); // Store the wet weight
136 float CalculatedDryWeight = WeightSensor->state - EvaporationTargetWeight->state; // Calculate next watering weight
137 if (CalculatedDryWeight >= StartWateringWeight->state)
138 DryWeight->publish_state(CalculatedDryWeight);
139 else
140 DryWeight->publish_state(StartWateringWeight->state);
141 DrainProgress = 0; // Reset the drain tracker
142 update_state(HempyStates::IDLE, true);
143 }
144 else
145 {
146 update_state(HempyStates::WATERING, true);
147 }
148 BlockOverWritingState = true;
149 }
150 break;
151 }
152
153 if (State != NewState && !BlockOverWritingState)
154 {
155 State = NewState;
156 }
157 UpdateInProgress = false;
158 }

247 float HempyBucket::update_average(float NewValue)
248 {
249 if (isnan(NewValue) || isinf(NewValue))
250 {
251 ESP_LOGE("hempy", "Invalid sensor reading detected: %.2f, skipping it.", NewValue);
252 return AverageWeight; // Keep the last valid average and skip this invalid value
253 }
254 else
255 {
256 if (AverageReset)
257 {
258 for (int i = 0; i < AverageQueueSize; ++i)
259 {
260 AverageReadings[i] = NewValue;
261 }
262 AverageTotal = AverageQueueSize * NewValue;
263 AverageReset = false;
264 }
265 else
266 {
267 AverageTotal += NewValue - AverageReadings[AverageCurrent]; // Add the difference of the new and previous reading to AverageTotal
268 AverageReadings[AverageCurrent] = NewValue;
269 }
270 AverageCurrent = (AverageCurrent + 1) % AverageQueueSize;
271
272 //ESP_LOGD("hempy", "Debug: NewValue=%.2f, AverageWeight=%.2f, AverageTotal=%.2f, AverageQueueSize=%d, AverageCurrent=%d,AverageReset=%d, Readings=[%.2f, %.2f, %.2f, %.2f, %.2f]",
273 // NewValue, AverageWeight, AverageTotal, AverageQueueSize, AverageCurrent, AverageReset,
274 // AverageReadings[0], AverageReadings[1], AverageReadings[2], AverageReadings[3], AverageReadings[4]);
275 return AverageTotal / AverageQueueSize;
276 }
