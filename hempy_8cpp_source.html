39 void HempyBucket::update_state(HempyStates NewState, bool Force)
40 {
41 if (UpdateInProgress && !Force)
42 {
43 return; // Exit if an update is already in progress
44 }
45 UpdateInProgress = true;
46 bool BlockOverWritingState = false; // Used when a state transitions to a new state
47 uint32_t CurrentTime = millis();
48 if (State != NewState)
49 {
50 StateTimer = CurrentTime; // Start measuring the time spent in the new State
51 ESP_LOGI("hempy", "%s State: %s -> %s", Name.c_str(), to_text_state(State), to_text_state(NewState));
52 }
53
54 switch (NewState)
55 {
56 case HempyStates::DISABLED:
57 if (WaterPump->state)
58 WaterPump->turn_off();
59 if (State != NewState) // When the state just changed
60 update_interval(DefaultUpdateInterval); // Restore the original update_interval from YAML
61 break;
62 case HempyStates::IDLE:
63 if (WaterPump->state)
64 WaterPump->turn_off();
65 if (State != NewState) // When the state just changed
66 update_interval(DefaultUpdateInterval); // Restore the original update_interval from YAML
67 if ((StartWateringWeight->state > 0 && WeightSensor->state <= StartWateringWeight->state) || (EvaporationTargetWeight->state > 0 && DryWeight->state > 0 && WeightSensor->state <= DryWeight->state))
68 {
69 update_state(HempyStates::WATERING, true);
70 BlockOverWritingState = true;
71 }
72 break;
73 case HempyStates::WATERING:
74 if (State != NewState)
75 {
76 update_interval(1000); // 1sec - Speed up calling update()
77 StateWeight = WeightSensor->state; // Store the bucket weight before starting the pump
78 if (State == HempyStates::IDLE || State == HempyStates::DISABLED) // First time entering the WATERING-DRAINING cycles
79 {
80 PumpOnTimer = CurrentTime;
81 WateringTimer = 0;
82 }
83 if (State == HempyStates::DRAINING)
84 {
85 PumpOnTimer = CurrentTime;
86 }
87 WaterPump->turn_on();
88 }
89 if (WeightSensor->state >= MaxWateringWeight->state || (WeightSensor->state >= StateWeight + WateringIncrement->state && WeightSensor->state >= WetWeight->state)) // Max weight reached OR (WateringIncrement's worth of water was added AND WetWeight is reached), wait for it to drain
90 {
91 WateringTimer += CurrentTime - PumpOnTimer;
92 update_state(HempyStates::DRAINING, true);
93 BlockOverWritingState = true;
94 }
95 else if ((CurrentTime - PumpOnTimer) > MaxWateringTime->state * 1000) // Disable watering in case MaxWateringTime is reached: Consider the pump broken
96 {
97 ESP_LOGW("Hempy", "%s Timeout, pump DISABLED", Name.c_str());
98 update_state(HempyStates::DISABLED, true);
99 BlockOverWritingState = true;
100 }
101 break;
102 case HempyStates::DRAINING:
103 if (WaterPump->state) // If the pump is on -> turn it off
104 WaterPump->turn_off();
105 if (State != NewState)
106 {
107 StateWeight = WeightSensor->state;
108 // ESP_LOGW("hempy", "Stored drain start weight: %.2f", StateWeight);
109 }
110 if (CurrentTime - StateTimer >= (DrainWaitTime->state * 1000)) // Waiting for the water to drain
111 {
112 DrainProgress += StateWeight - WeightSensor->state; // Calculate how much water has drained
113 // ESP_LOGW("hempy", "if %.2f <= %.2f - %.2f ", WeightSensor->state, StateWeight, DrainTargetWeight->state);
114 if (DrainProgress >= DrainTargetWeight->state || WeightSensor->state >= MaxWateringWeight->state) // Check if enough water was drained into the waste reservoir, OR MaxWateringWeight is reached
115 {
116 WetWeight->publish_state(WeightSensor->state); // Store the wet weight
117 float CalculatedDryWeight = WeightSensor->state - EvaporationTargetWeight->state; // Calculate next watering weight
118 if (CalculatedDryWeight >= StartWateringWeight->state)
119 DryWeight->publish_state(CalculatedDryWeight);
120 else
121 DryWeight->publish_state(StartWateringWeight->state);
122 DrainProgress = 0; // Reset the drain tracker
123 update_state(HempyStates::IDLE, true);
124 }
125 else
126 {
127 update_state(HempyStates::WATERING, true);
128 }
129 BlockOverWritingState = true;
130 }
131 break;
132 }
133
134 if (State != NewState && !BlockOverWritingState)
135 {
136 State = NewState;
137 }
138 UpdateInProgress = false;
139 }
