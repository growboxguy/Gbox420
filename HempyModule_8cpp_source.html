296 void HempyModule::updateAckData(HempyMessages NewSequenceID)
297 {
298  //Wireless.flush_tx(); ///< Dump all previously cached but unsent ACK messages from the TX FIFO buffer (Max 3 are saved)
299  NextSequenceID = NewSequenceID; // update the next Message that will be copied to the buffer
300 
301  switch (NextSequenceID) // based on the NextSeqenceID load the next response into the Acknowledgement buffer
302  {
303  case HempyMessages::HempyModuleResponse1:
304  Wireless.writeAckPayload(1, &HempyModuleResponse1ToSend, WirelessPayloadSize);
305  break;
306  case HempyMessages::HempyBucketResponse1:
307  Wireless.writeAckPayload(1, &HempyBucket1ResponseToSend, WirelessPayloadSize);
308  break;
309  case HempyMessages::HempyBucketResponse2:
310  Wireless.writeAckPayload(1, &HempyBucket2ResponseToSend, WirelessPayloadSize);
311  break;
312  case HempyMessages::HempyReset:
313  Wireless.writeAckPayload(1, &HempyResetToSend, WirelessPayloadSize);
314  break;
315  default:
316  Wireless.writeAckPayload(1, &HempyModuleResponse1ToSend, WirelessPayloadSize); // load the first Response into the buffer
317  break;
318  }
