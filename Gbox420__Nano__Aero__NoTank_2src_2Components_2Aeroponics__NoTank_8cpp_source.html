61 void Aeroponics_NoTank::updateState(AeroNoTankStates NewState)
62 {
63  bool BlockOverWritingState = false; //Used when a state transitions to a new state
64  if (State != NewState)
65  {
66  memset(&LongMessage[0], 0, sizeof(LongMessage));
67  strcat_P(LongMessage, (PGM_P)Name);
68  strcat_P(LongMessage, (PGM_P)F(" state: "));
69  strcat(LongMessage, toText_aeroNoTankState(State));
70  strcat_P(LongMessage, (PGM_P)F(" -> "));
71  strcat(LongMessage, toText_aeroNoTankState(NewState));
72  logToSerials(&LongMessage, true, 3);
73  }
74 
75  switch (NewState)
76  {
77  case AeroNoTankStates::DISABLED:
78  if (State != NewState)
79  {
80  Pump->stopPump();
81  }
82  break;
83  case AeroNoTankStates::IDLE:
84  if (State != NewState)
85  {
86  Pump->stopPump();
87  }
88  uint32_t Interval;
89  if (DayMode)
90  {
91  Interval = *DayInterval * 60000;
92  }
93  else
94  {
95  Interval = *NightInterval * 60000;
96  }
97  if (millis() - SprayTimer >= Interval)
98  {
99  updateState(AeroNoTankStates::PRIMING);
100  BlockOverWritingState = true;
101  }
102  break;
103  case AeroNoTankStates::PRIMING:
104  if (State != NewState)
105  {
106  Pump->startPump(true);
107  }
108  if (Pump->getState() != PressurePumpStates::PRIMING)
109  {
110  updateState(AeroNoTankStates::SPRAY);
111  BlockOverWritingState = true;
112  }
113  break;
114  case AeroNoTankStates::SPRAY:
115  if (State != NewState)
116  {
117  SprayTimer = millis();
118  }
119  if (FeedbackPressureSensor->readPressure(false) > *MaxPressure || millis() - SprayTimer >= *Duration * 1000)
120  {
121  LastSprayPressure = FeedbackPressureSensor->getPressure();
122  updateState(AeroNoTankStates::RELEASE);
123  BlockOverWritingState = true;
124  }
125  break;
126  case AeroNoTankStates::RELEASE: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
127  if (State != NewState)
128  {
129  Pump->stopPump();
130  }
131  if (Pump->getState() == PressurePumpStates::IDLE)
132  {
133  updateState(AeroNoTankStates::IDLE);
134  BlockOverWritingState = true;
135  }
136  if (Pump->getState() == PressurePumpStates::DISABLED)
137  {
138  updateState(AeroNoTankStates::DISABLED);
139  BlockOverWritingState = true;
140  }
141  break;
142  case AeroNoTankStates::MIX: //< Release remaining pressure from the misting loop AND pump output side
143  if (State != NewState)
144  {
145  Pump->startMixing();
146  }
147  if (Pump->getState() != PressurePumpStates::MIX)
148  {
149  updateState(AeroNoTankStates::IDLE);
150  BlockOverWritingState = true;
151  }
152  break;
153  }
154 
155  if (State != NewState && !BlockOverWritingState)
156  {
157  State = NewState;
158  }
159 }
