70 void Aeroponics_Tank::updateState(AeroTankStates NewState)
71 {
72  bool BlockOverWritingState = false; //Used when a state transitions to a new state
73  if (State != NewState)
74  {
75  memset(&LongMessage[0], 0, sizeof(LongMessage));
76  strcat_P(LongMessage, (PGM_P)Name);
77  strcat_P(LongMessage, (PGM_P)F(" state: "));
78  strcat(LongMessage, toText_aeroTankState(State));
79  strcat_P(LongMessage, (PGM_P)F(" -> "));
80  strcat(LongMessage, toText_aeroTankState(NewState));
81  logToSerials(&LongMessage, true, 3);
82  }
83 
84  switch (NewState)
85  {
86  case AeroTankStates::DISABLED:
87  if (State != NewState)
88  {
89  *SprayEnabled = false;
90  SpraySwitch->turnOff();
91  Pump->stopPump();
92  }
93  break;
94  case AeroTankStates::IDLE:
95  if (State != NewState)
96  {
97  *SprayEnabled = true;
98  SpraySwitch->turnOff();
99  Pump->stopPump();
100  }
101  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
102  {
103  uint32_t Interval;
104  if (DayMode)
105  Interval = *DayInterval * 60000;
106  else
107  Interval = *NightInterval * 60000;
108  if (millis() - SprayTimer >= Interval)
109  {
110  updateState(AeroTankStates::SPRAY);
111  BlockOverWritingState = true;
112  }
113  }
114  if (Pump->getState() == PressurePumpStates::IDLE && FeedbackPressureSensor->getPressure() <= *MinPressure)
115  {
116  logToSerials(F("Tank recharging"), false, 3);
117  updateState(AeroTankStates::REFILL);
118  BlockOverWritingState = true;
119  }
120  break;
121  case AeroTankStates::SPRAY:
122  if (State != NewState)
123  {
124  *SprayEnabled = true;
125  SpraySwitch->turnOn();
126  SprayTimer = millis();
127  }
128  if (millis() - SprayTimer >= *Duration * 1000)
129  {
130  LastSprayPressure = Aeroponics_Tank::FeedbackPressureSensor->getPressure();
131  updateState(AeroTankStates::STOPSPRAY);
132  BlockOverWritingState = true;
133  }
134  break;
135  case AeroTankStates::STOPSPRAY: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
136  if (State != NewState)
137  {
138  SpraySwitch->turnOff();
139  SprayTimer = millis();
140  }
141  if (millis() - SprayTimer >= *SpraySolenoidClosingDelay)
142  {
143  SprayTimer = millis();
144  updateState(AeroTankStates::RELEASE);
145  BlockOverWritingState = true;
146  }
147  break;
148  case AeroTankStates::RELEASE: //< Release remaining pressure from the misting loop AND pump output side
149  if (State != NewState)
150  {
151  Pump->startBlowOff();
152  }
153  if (Pump->getState() != PressurePumpStates::BLOWOFF && Pump->getState() != PressurePumpStates::BYPASSCLOSE)
154  {
155  updateState(AeroTankStates::IDLE);
156  BlockOverWritingState = true;
157  }
158  break;
159  case AeroTankStates::REFILL:
160  if (State != NewState)
161  {
162  Pump->startPump(true);
163  }
164  if (FeedbackPressureSensor->readPressure(false) >= *MaxPressure)
165  {
166  logToSerials(F("Tank recharged"), true, 3);
167  updateState(AeroTankStates::RELEASE);
168  BlockOverWritingState = true;
169  }
170  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
171  {
172  logToSerials(F("Recharge failed"), true, 3);
173  updateState(AeroTankStates::IDLE);
174  BlockOverWritingState = true;
175  }
176  break;
177  case AeroTankStates::DRAIN:
178  if (State != NewState)
179  {
180  Pump->turnBypassOn(true);
181  SpraySwitch->turnOn();
182  }
183  if (FeedbackPressureSensor->readPressure(false) <= 0.1 || (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED))
184  {
185  if (Pump->getState() == PressurePumpStates::BYPASSOPEN)
186  Pump->turnBypassOff();
187  SpraySwitch->turnOff();
188  logToSerials(F("Tank drained"), true, 3);
189  if (*SprayEnabled)
190  {
191  updateState(AeroTankStates::IDLE);
192  }
193  else
194  {
195  updateState(AeroTankStates::DISABLED);
196  }
197  BlockOverWritingState = true;
198  }
199  break;
200  case AeroTankStates::MIX:
201  if (State != NewState)
202  {
203  Pump->startMixing();
204  }
205  if (Pump->getState() != PressurePumpStates::MIX)
206  {
207  updateState(AeroTankStates::IDLE);
208  BlockOverWritingState = true;
209  }
210  break;
211  }
212 
213  if (State != NewState && !BlockOverWritingState)
214  {
215  State = NewState;
216  }
