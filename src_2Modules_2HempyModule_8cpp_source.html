273 void HempyModule::updateAckData(HempyMessages NewSequenceID)
274 {
275  //Wireless.flush_tx(); ///< Dump all previously cached but unsent ACK messages from the TX FIFO buffer (Max 3 are saved)
276  NextSequenceID = NewSequenceID; // update the next Message that will be copied to the buffer
277 
278  switch (NextSequenceID) // based on the NextSeqenceID load the next response into the Acknowledgement buffer
279  {
280  case HempyMessages::HempyModuleResponse1:
281  Wireless.writeAckPayload(1, &HempyModuleResponse1ToSend, WirelessPayloadSize);
282  break;
283  case HempyMessages::HempyBucket1Response1:
284  Wireless.writeAckPayload(1, &HempyBucket1Response1ToSend, WirelessPayloadSize);
285  break;
286  case HempyMessages::HempyBucket2Response1:
287  Wireless.writeAckPayload(1, &HempyBucket2Response1ToSend, WirelessPayloadSize);
288  break;
289  case HempyMessages::HempyReset:
290  Wireless.writeAckPayload(1, &HempyResetToSend, WirelessPayloadSize);
291  break;
292  default:
293  Wireless.writeAckPayload(1, &HempyModuleResponse1ToSend, WirelessPayloadSize); // load the first Response into the buffer
294  break;
295  }
