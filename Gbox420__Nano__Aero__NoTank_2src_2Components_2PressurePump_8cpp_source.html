52 {

54  if (State != NewState)
55  {
56  memset(&LongMessage[0], 0, sizeof(LongMessage));
57  strcat_P(LongMessage, (PGM_P)Name);
58  strcat_P(LongMessage, (PGM_P)F(" state: "));
59  strcat(LongMessage, toText_pressurePumpState(State));
60  strcat_P(LongMessage, (PGM_P)F(" -> "));

63 
64  PumpTimer = millis();
65  }
66 
67  switch (NewState)
68  {
69  case PressurePumpStates::IDLE:
70  if (State != NewState)
71  {
72  PumpSwitch->turnOff();
73  BypassSwitch->turnOff();
74  *PumpEnabled = true;
75  }
76  break;
77  case PressurePumpStates::PRIMING:
78  if (State != NewState)
79  {
80  PumpSwitch->turnOn();
81  BypassSwitch->turnOn();
82  }
83  if (millis() - PumpTimer > ((uint32_t)*PrimingTime * 1000))
84  {
85  updateState(PressurePumpStates::RUNNING);
86  BlockOverWritingState = true;
87  }
88  break;
89  case PressurePumpStates::RUNNING:
90  if (State != NewState)
91  {
92  PumpSwitch->turnOn();
93  BypassSwitch->turnOff();
94  *PumpEnabled = true;
95  }
96  if (millis() - PumpTimer > ((uint32_t)*PumpTimeOut * 1000))
97  {
98  Parent->addToLog(F("Pump timeout"), 3);
99  *PumpEnabled = false;
100  updateState(PressurePumpStates::BLOWOFF);
101  BlockOverWritingState = true;
102  }
103  break;
104  case PressurePumpStates::BLOWOFF:
105  if (State != NewState)
106  {
107  PumpSwitch->turnOff();
108  BypassSwitch->turnOn();
109  }
110  if (millis() - PumpTimer > ((uint32_t)*BlowOffTime * 1000))
111  {
112  updateState(PressurePumpStates::BYPASSCLOSE);
113  BlockOverWritingState = true;
114  }
115  break;
116  case PressurePumpStates::BYPASSOPEN:
117  if (State != NewState)
118  {
119  PumpSwitch->turnOff();
120  BypassSwitch->turnOn();
121  }
122  if (millis() - PumpTimer > ((uint32_t)*BypassSolenoidMaxOpenTime * 1000))
123  {
124  updateState(PressurePumpStates::BYPASSCLOSE);
125  BlockOverWritingState = true;
126  }
127  break;
128  case PressurePumpStates::BYPASSCLOSE:
129  if (State != NewState)
130  {
131  PumpSwitch->turnOff();
132  BypassSwitch->turnOff();
133  }
134  if (millis() - PumpTimer > ((uint32_t)*BypassSolenoidClosingDelay))
135  {
136  if (*PumpEnabled)
137  updateState(PressurePumpStates::IDLE);
138  else
139  updateState(PressurePumpStates::DISABLED);
140  BlockOverWritingState = true;
141  }
142  break;
143  case PressurePumpStates::MIX:
144  if (State != NewState)
145  {
146  PumpSwitch->turnOn();
147  BypassSwitch->turnOn();
148  }
149  *PumpEnabled = true;
150  if (millis() - PumpTimer > ((uint32_t)*PumpTimeOut * 1000))
151  {
152  updateState(PressurePumpStates::BYPASSCLOSE);
153  BlockOverWritingState = true;
154  }
155  break;
156  case PressurePumpStates::DISABLED:
157  if (State != NewState)
158  {
159  PumpSwitch->turnOff();
160  BypassSwitch->turnOff();
161  *PumpEnabled = false;
162  }
163  break;
164  }
