107 void Lights::checkDimming()
108 {
109  if (CurrentStatus == LightStates::FADEIN)
110  {
111  if (millis() - FadingTimer >= *FadingInterval * 1000)
112  {
113  CurrentBrightness = CurrentBrightness + *FadingIncrements; //Increase current brightness
114  if (CurrentBrightness >= *Brightness)
115  {
116  CurrentBrightness = *Brightness;
117  CurrentStatus = LightStates::TURNEDON;
118  }
119  setBrightness(CurrentBrightness, false, false);
120  FadingTimer = millis();
121  }
122  }
123  else if (CurrentStatus == LightStates::FADEOUT)
124  {
125  if (millis() - FadingTimer >= *FadingInterval * 1000)
126  {
127  CurrentBrightness = CurrentBrightness - *FadingIncrements; //Decrease current brightness
128  if (CurrentBrightness <= 0)
129  {
130  CurrentBrightness = 0;
131  CurrentStatus = LightStates::TURNEDOFF;
132  }
133  setBrightness(CurrentBrightness, false, false);
134  FadingTimer = millis();
135  }
136  }
137  else if (CurrentStatus == LightStates::DIMMED)
138  {
139  if (millis() - DimmingTimer >= *DimmingDuration * 60000)
140  {
141  dimLightsOnOff();
142  }
143  }
144  checkRelay();

147 void Lights::checkTimer()
148 {
149  if (*TimerEnabled)
150  {
151  time_t Now = now(); // Get the current time from TimeLib
152  int CombinedOnTime = *OnHour * 100 + *OnMinute; // convert time to number, Example: 8:10=810, 20:10=2010
153  int CombinedOffTime = *OffHour * 100 + *OffMinute;
154  int CombinedCurrentTime = hour(Now) * 100 + minute(Now);
155  if (CombinedOnTime <= CombinedOffTime) // no midnight turnover, Example: On 8:10, Off: 20:10
156  {
157  if (CombinedOnTime <= CombinedCurrentTime && CombinedCurrentTime < CombinedOffTime)
158  { // True: Light should be on
159  if (!*Status)
160  {
161  setLightOnOff(true, false); // If status is OFF: Turn ON the lights (First bool), and do not add it to the log (Second bool)
162  appendName(true);
163  strcat_P(ShortMessage, (PGM_P)F("timer: ON"));
164  logToSerials(&ShortMessage);
165  }
166  }
167  else // False: Light should be off
168  {
169  if (*Status)
170  { // If status is ON
171  setLightOnOff(false, false); // Turn OFF the lights (First bool), and do not add it to the log (Second bool)
172  appendName(true);
173  strcat_P(ShortMessage, (PGM_P)F("timer: OFF"));
174  logToSerials(&ShortMessage);
175  }
176  }
177  }
178  else // midnight turnover, Example: On 21:20, Off: 9:20
179  {
180  if (CombinedOnTime <= CombinedCurrentTime || CombinedCurrentTime < CombinedOffTime)
181  {
182  if (!*Status)
183  {
184  setLightOnOff(true, false);
185  appendName(true);
186  strcat_P(ShortMessage, (PGM_P)F("timer ON"));
187  logToSerials(&ShortMessage);
188  }
189  }
190  else if (*Status)
191  {
192  setLightOnOff(false, false);
193  appendName(true);
194  strcat_P(ShortMessage, (PGM_P)F("timer OFF"));
195  logToSerials(&ShortMessage);
196  }
