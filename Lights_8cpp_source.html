111  {

113  if (CurrentBrightness >= Brightness)

146 void Lights::checkTimer()
147 {
148  if (TimerEnabled)
149  {
150  time_t Now = now(); // Get the current time from TimeLib
151  int CombinedOnTime = OnHour * 100 + OnMinute; // convert time to number, Example: 8:10=810, 20:10=2010
152  int CombinedOffTime = OffHour * 100 + OffMinute;
153  int CombinedCurrentTime = hour(Now) * 100 + minute(Now);
154  if (CombinedOnTime <= CombinedOffTime) // no midnight turnover, Example: On 8:10, Off: 20:10
155  {
156  if (CombinedOnTime <= CombinedCurrentTime && CombinedCurrentTime < CombinedOffTime)
157  { // True: Light should be on
158  if (!Status)
159  {
160  setLightOnOff(true, true); // If status is OFF: Turn ON the lights (First bool), and do not add it to the log (Second bool)
161  }
162  }
163  else // False: Light should be off
164  {
165  if (Status)
166  { // If status is ON
167  setLightOnOff(false, true); // Turn OFF the lights (First bool), and do not add it to the log (Second bool)
168  }
169  }
170  }
171  else // midnight turnover, Example: On 21:20, Off: 9:20
172  {
173  if (CombinedOnTime <= CombinedCurrentTime || CombinedCurrentTime < CombinedOffTime)
174  {
175  if (!Status)
176  {
177  setLightOnOff(true, true);
178  }
179  }
180  else if (Status)
181  {
182  setLightOnOff(false, true);
183  }
184  }
