303 // NTP data received
304 void ntpReceived(void *arg, struct udp_pcb *pcb, struct pbuf *p, const ip_addr_t *addr, uint16_t port)
305 {
306  uint8_t mode = pbuf_get_at(p, 0) & 0x7;
307  uint8_t stratum = pbuf_get_at(p, 1);
308  datetime_t timeTemp;
309  char CurrentTimeText[MaxShotTextLength];
310  // Check the result
311  if (ip_addr_cmp(addr, &NtpServerIP) && port == GboxSettings->NtpServer1.NtpServerPort && p->tot_len == 48 && mode == 0x4 && stratum != 0)
312  {
313  uint8_t seconds_buf[4] = {0};
314  pbuf_copy_partial(p, seconds_buf, sizeof(seconds_buf), 40);
315  uint32_t seconds_since_1900 = seconds_buf[0] << 24 | seconds_buf[1] << 16 | seconds_buf[2] << 8 | seconds_buf[3];
316  uint32_t seconds_since_1970 = seconds_since_1900 - 2208988800; // 2208988800 seconds between 1 Jan 1900 and 1 Jan 1970
317  time_t epoch = seconds_since_1970;
318  struct tm *utc = gmtime(&epoch); // https://cplusplus.com/reference/ctime/tm/
319  timeTemp = {
320  .year = (int16_t)(utc->tm_year + 1900), // utc->tm_year: years since 1900
321  .month = (int8_t)(utc->tm_mon + 1), // utc->tm_mon: months since January (0-11)
322  .day = (int8_t)utc->tm_mday,
323  .dotw = (int8_t)utc->tm_wday, // utc->tm_wday: days since Sunday (0-6)
324  .hour = (int8_t)(utc->tm_hour + GboxSettings->NtpServer1.TimeZoneDifference),
325  .min = (int8_t)utc->tm_min,
326  .sec = (int8_t)utc->tm_sec};
327  datetime_to_str(CurrentTimeText, sizeof(CurrentTimeText), &timeTemp);
328  if (rtc_set_datetime(&timeTemp))
329  {
330  printf("RTC synced: %s\n", CurrentTimeText);
331  NtpSynced = 1;
332  }
333  else
334  printf("RTC update failed\n");
335  }
336  else
337  {
338  printf("Invalid NTP response\n");
339  }
340  pbuf_free(p);
341 }
