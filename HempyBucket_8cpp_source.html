60 void HempyBucket::updateState(HempyStates NewState)
61 {
62  bool BlockOverWritingState = false; //Used when a state transitions to a new state
63  if (State != NewState)
64  {
65  StateTimer = millis();
66  memset(&LongMessage[0], 0, MaxLongTextLength);
67  strcat_P(LongMessage, (PGM_P)Name);
68  strcat_P(LongMessage, (PGM_P)F(" state: "));
69  strcat(LongMessage, toText_hempyState(State));
70  strcat_P(LongMessage, (PGM_P)F(" -> "));
71  strcat(LongMessage, toText_hempyState(NewState));
72  logToSerials(&LongMessage, true, 3);
73  }
74 
75  BucketWeightSensor->readWeight(false);
76  WasteReservoirWeightSensor->readWeight(false);
77 
78  switch (NewState)
79  {
80  case HempyStates::DISABLED:
81  BucketPump->disablePump();
82  break;
83  case HempyStates::IDLE:
84  BucketPump->stopPump(true);
85  if (BucketPump->getState() != WaterPumpStates::DISABLED && (DryWeight <= 0 || BucketWeightSensor->getWeight() <= DryWeight))
86  {
87  if (BucketWeightSensor->getWeight() >= DryWeight - *OverflowTarget)
88  {
89  updateState(HempyStates::WATERING);
90  BlockOverWritingState = true;
91  }
92  }
93  break;
94  case HempyStates::WATERING:
95  if (State != NewState)
96  {
97  BucketStartWeight = BucketWeightSensor->getWeight(); // Store the bucket weight before starting the pump
98  if (State == HempyStates::IDLE || State == HempyStates::DISABLED) // First time entering the WATERING-DRIAINING cycles
99  {
100  WasteReservoirStartWeight = WasteReservoirWeightSensor->getWeight(); //Store the waste reservoir weight, watering will stop once WasteReservoirStartWeight + OverflowTarget is reached
101  PumpOnTimer = millis();
102  WateringTime = 0;
103  }
104  if (State == HempyStates::DRAINING)
105  {
106  PumpOnTimer = millis();
107  }
108  BucketPump->startPump(true);
109  }
110  if (BucketWeightSensor->getWeight(false) >= WetWeight && BucketWeightSensor->getWeight(false) - BucketStartWeight + WasteReservoirWeightSensor->getWeight() - WasteReservoirStartWeight >= *OverflowTarget) //Wet weight reached AND Target overflow's worth of water was added, wait for it to drain
111  {
112  WateringTime += millis() - PumpOnTimer;
113  updateState(HempyStates::DRAINING);
114  BlockOverWritingState = true;
115  }
116  if (WateringTime > ((uint32_t)BucketPump->getPumpTimeOut() * 1000) || BucketPump->getState() == WaterPumpStates::DISABLED || WasteReservoirWeightSensor->getWeight() >= *WasteLimit)
117  {
118  updateState(HempyStates::DISABLED);
119  BlockOverWritingState = true;
120  }
121  break;
122  case HempyStates::DRAINING:
123  BucketPump->stopPump();
124  State = HempyStates::DRAINING; //Store the new state immediately - Only important when DrainWaitTime is set to 0
125  if (millis() - StateTimer > ((uint32_t)*DrainWaitTime * 1000))
126  {
127  if (WasteReservoirWeightSensor->getWeight(false) - WasteReservoirStartWeight >= *OverflowTarget) //Check if target overflow weight is reached
128  {
129  WetWeight = BucketWeightSensor->getWeight(); //Measure wet weight
130  DryWeight = WetWeight - *EvaporationTarget; //Calculate next watering weight
131  updateState(HempyStates::IDLE);
132  }
133  else
134  {
135  if (WasteReservoirWeightSensor->getWeight() >= *WasteLimit)
136  {
137  updateState(HempyStates::DISABLED);
138  }
139  else
140  {
141  updateState(HempyStates::WATERING);
142  }
143  }
144  BlockOverWritingState = true;
145  }
146  break;
147  }
148 
149  if (State != NewState && !BlockOverWritingState)
150  {
151  State = NewState;
152  }
153 }
