77 void HempyBucket::updateState(HempyStates NewState)
78 {
79  bool BlockOverWritingState = false; // Used when a state transitions to a new state
80  if (State != NewState)
81  {
82  StateTimer = millis();
83  memset(&LongMessage[0], 0, MaxLongTextLength);
84  strcat(LongMessage, getName(F("state: ")));
85  strcat(LongMessage, toText_hempyState(State));
86  strcat_P(LongMessage, (PGM_P)F(" -> "));
87  strcat(LongMessage, toText_hempyState(NewState));
88  logToSerials(&LongMessage, true, 3);
89  }
90 
91  BucketWeightSensor.readWeight();
92 
93  switch (NewState)
94  {
95  case HempyStates::DISABLED:
96  BucketPump.disablePump();
97  break;
98  case HempyStates::IDLE:
99  BucketPump.stopPump(true);
100  if (BucketPump.getState() != WaterPumpStates::DISABLED && BucketWeightSensor.getWeight() <= DryWeight)
101  {
102  updateState(HempyStates::WATERING);
103  BlockOverWritingState = true;
104  }
105  break;
106  case HempyStates::WATERING:
107  if (State != NewState)
108  {
109  StateWeight = BucketWeightSensor.getWeight(); // Store the bucket weight before starting the pump
110  if (State == HempyStates::IDLE || State == HempyStates::DISABLED) // First time entering the WATERING-DRIAINING cycles
111  {
112  PumpOnTimer = millis();
113  WateringTime = 0;
114  }
115  if (State == HempyStates::DRAINING)
116  {
117  PumpOnTimer = millis();
118  }
119  BucketPump.startPump(true);
120  }
121  if (BucketWeightSensor.getWeight() >= StateWeight + WateringIncrement && BucketWeightSensor.getWeight() >= WetWeight)
122  {
123  WateringTime += millis() - PumpOnTimer;
124  updateState(HempyStates::DRAINING);
125  BlockOverWritingState = true;
126  }
127  if (WateringTime > ((uint32_t)BucketPump.getTimeOut() * 1000) || BucketPump.getState() == WaterPumpStates::DISABLED || BucketWeightSensor.getWeight() > MaxWeight)
128  {
129  updateState(HempyStates::DISABLED);
130  BlockOverWritingState = true;
131  }
132  break;
133  case HempyStates::DRAINING:
134  BucketPump.stopPump();
135  if (State != NewState)
136  {
137  StateWeight = BucketWeightSensor.getWeight();
138  }
139  if (millis() - StateTimer > ((uint32_t)DrainWaitTime * 1000))
140  {
141  DrainProgress += StateWeight - BucketWeightSensor.getWeight();
142  /*if(Debug)
143  {
144  logToSerials(F("Drained:"), false, 3);
145  logToSerials(DrainProgress,false, 1);
146  logToSerials(F("/"), false, 0);
147  logToSerials(DrainTargetWeight,true, 0);
148  }
149  */
150  if (DrainProgress >= DrainTargetWeight) // Check if target overflow weight is reached -> Watering completed
151  {
152  WetWeight = BucketWeightSensor.getWeight();
153  DryWeight = WetWeight - EvaporationTarget;
154  if (DryWeight < StartWeight)
155  DryWeight = StartWeight;
156  DrainProgress = 0;
157  updateState(HempyStates::IDLE);
158  }
159  else
160  {
161  updateState(HempyStates::WATERING);
162  }
163  BlockOverWritingState = true;
164  }
165  break;
166  }
167 
168  if (State != NewState && !BlockOverWritingState)
169  {
170  State = NewState;
171  }
172 }
