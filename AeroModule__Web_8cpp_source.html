323  break;
324  case AeroMessages::AeroResponse1:

362  logToSerials(AeroResponse2Received.ConfirmDrainPressureTank, false, 1);
363  logToSerials(AeroResponse2Received.ConfirmTareWeight, true, 1);
364  }
365  if (AeroCommand2ToSend.PumpOn || AeroCommand2ToSend.PumpOff || AeroCommand2ToSend.PumpDisable || AeroCommand2ToSend.MixReservoir || AeroCommand2ToSend.RefillPressureTank || AeroCommand2ToSend.DrainPressureTank || AeroCommand2ToSend.TareWeight)
366  {
