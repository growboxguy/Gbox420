245 void AeroModule::updateAckData(AeroMessages NewSequenceID)
246 {
247  //Wireless.flush_tx(); ///< Dump all previously cached but unsent ACK messages from the TX FIFO buffer (Max 3 are saved)
248  NextSequenceID = NewSequenceID; // update the next Message that will be copied to the buffer
249 
250  switch (NextSequenceID) // based on the NextSeqenceID load the next response into the Acknowledgement buffer
251  {
252  case AeroMessages::AeroModuleResponse1:
253  Wireless.writeAckPayload(1, &AeroModule1ResponseToSend, WirelessPayloadSize);
254  break;
255  case AeroMessages::AeroResponse1:
256  Wireless.writeAckPayload(1, &Aero1Response1ToSend, WirelessPayloadSize);
257  break;
258  case AeroMessages::AeroResponse2:
259  Wireless.writeAckPayload(1, &Aero1Response2ToSend, WirelessPayloadSize);
260  break;
261  case AeroMessages::AeroReset:
262  Wireless.writeAckPayload(1, &AeroResetToSend, WirelessPayloadSize);
263  break;
264  default:
265  Wireless.writeAckPayload(1, &AeroModule1ResponseToSend, WirelessPayloadSize); // load the first Response into the buffer
266  break;
