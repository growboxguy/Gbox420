73 void Aeroponics_Tank::updateState(AeroTankStates NewState)
74 {
75  bool BlockOverWritingState = false; //Used when a state transitions to a new state
76  if (State != NewState)
77  {
78  memset(&LongMessage[0], 0, MaxLongTextLength);
79  strcat_P(LongMessage, (PGM_P)Name);
80  strcat_P(LongMessage, (PGM_P)F(" state: "));
81  strcat(LongMessage, toText_aeroTankState(State));
82  strcat_P(LongMessage, (PGM_P)F(" -> "));
83  strcat(LongMessage, toText_aeroTankState(NewState));
84  logToSerials(&LongMessage, true, 3);
85  }
86 
87  switch (NewState)
88  {
89  case AeroTankStates::DISABLED:
90  if (State != NewState)
91  {
92  *SprayEnabled = false;
93  SpraySwitch->turnOff();
94  Pump->stopPump();
95  }
96  break;
97  case AeroTankStates::IDLE:
98  if (State != NewState)
99  {
100  *SprayEnabled = true;
101  SpraySwitch->turnOff();
102  Pump->stopPump();
103  }
104  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
105  {
106  uint32_t Interval;
107  if (DayMode)
108  Interval = *DayInterval * 60000;
109  else
110  Interval = *NightInterval * 60000;
111  if (millis() - SprayTimer >= Interval)
112  {
113  updateState(AeroTankStates::SPRAY);
114  BlockOverWritingState = true;
115  }
116  }
117  if (Pump->getState() == PressurePumpStates::IDLE && FeedbackPressureSensor->getPressure() <= *MinPressure)
118  {
119  updateState(AeroTankStates::REFILL);
120  BlockOverWritingState = true;
121  }
122  break;
123  case AeroTankStates::SPRAY:
124  if (State != NewState)
125  {
126  *SprayEnabled = true;
127  SpraySwitch->turnOn();
128  SprayTimer = millis();
129  }
130  if (millis() - SprayTimer >= *Duration * 1000)
131  {
132  LastSprayPressure = Aeroponics_Tank::FeedbackPressureSensor->getPressure();
133  updateState(AeroTankStates::STOPSPRAY);
134  BlockOverWritingState = true;
135  }
136  break;
137  case AeroTankStates::STOPSPRAY: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
138  if (State != NewState)
139  {
140  SpraySwitch->turnOff();
141  SprayTimer = millis();
142  }
143  if (millis() - SprayTimer >= *SpraySolenoidClosingDelay)
144  {
145  SprayTimer = millis();
146  updateState(AeroTankStates::RELEASE);
147  BlockOverWritingState = true;
148  }
149  break;
150  case AeroTankStates::RELEASE: //< Release remaining pressure from the misting loop AND pump output side
151  if (State != NewState)
152  {
153  Pump->startBlowOff();
154  }
155  if (Pump->getState() != PressurePumpStates::BLOWOFF && Pump->getState() != PressurePumpStates::BYPASSCLOSE)
156  {
157  updateState(AeroTankStates::IDLE);
158  BlockOverWritingState = true;
159  }
160  break;
161  case AeroTankStates::REFILL:
162  if (State != NewState)
163  {
164  Pump->startPump(true);
165  }
166  if (FeedbackPressureSensor->readPressure(false) >= *MaxPressure)
167  {
168  updateState(AeroTankStates::RELEASE);
169  BlockOverWritingState = true;
170  }
171  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
172  {
173  //logToSerials(F("Recharge failed"), true, 3);
174  updateState(AeroTankStates::IDLE);
175  BlockOverWritingState = true;
176  }
177  break;
178  case AeroTankStates::DRAIN:
179  if (State != NewState)
180  {
181  Pump->turnBypassOn(true);
182  SpraySwitch->turnOn();
183  }
184  if (FeedbackPressureSensor->readPressure(false) <= 0.1 || (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED))
185  {
186  if (Pump->getState() == PressurePumpStates::BYPASSOPEN)
187  Pump->turnBypassOff();
188  SpraySwitch->turnOff();
189  if (*SprayEnabled)
190  {
191  updateState(AeroTankStates::IDLE);
192  }
193  else
194  {
195  updateState(AeroTankStates::DISABLED);
196  }
197  BlockOverWritingState = true;
198  }
199  break;
200  case AeroTankStates::MIX:
201  if (State != NewState)
202  {
203  Pump->startMixing();
204  }
205  if (Pump->getState() != PressurePumpStates::MIX)
206  {
207  updateState(AeroTankStates::IDLE);
208  BlockOverWritingState = true;
209  }
210  break;
211  }
212 
213  if (State != NewState && !BlockOverWritingState)
214  {
215  State = NewState;
216  }
