79 void Aeroponics_Tank::updateState(AeroTankStates NewState)
80 {
81  bool BlockOverWritingState = false; // Used when a state transitions to a new state
82  if (State != NewState)
83  {
84  memset(&LongMessage[0], 0, MaxLongTextLength);
85  strcat_P(LongMessage, (PGM_P)Name);
86  strcat_P(LongMessage, (PGM_P)F(" state: "));
87  strcat(LongMessage, toText_aeroTankState(State));
88  strcat_P(LongMessage, (PGM_P)F(" -> "));
89  strcat(LongMessage, toText_aeroTankState(NewState));
90  logToSerials(&LongMessage, true, 3);
91  }
92 
93  switch (NewState)
94  {
95  case AeroTankStates::DISABLED:
96  if (State != NewState)
97  {
98  SprayEnabled = false;
99  SpraySwitch->turnOff();
100  Pump->stopPump();
101  }
102  break;
103  case AeroTankStates::IDLE:
104  if (State != NewState)
105  {
106  SprayEnabled = true;
107  SpraySwitch->turnOff();
108  Pump->stopPump();
109  }
110  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
111  {
112  uint32_t Interval;
113  if (DayMode)
114  Interval = DayInterval * 60000;
115  else
116  Interval = NightInterval * 60000;
117  if (millis() - SprayTimer >= Interval)
118  {
119  updateState(AeroTankStates::SPRAY);
120  BlockOverWritingState = true;
121  }
122  }
123  if (Pump->getState() == PressurePumpStates::IDLE && FeedbackPressureSensor->getPressure() <= MinPressure)
124  {
125  updateState(AeroTankStates::REFILL);
126  BlockOverWritingState = true;
127  }
128  break;
129  case AeroTankStates::SPRAY:
130  if (State != NewState)
131  {
132  SprayEnabled = true;
133  SpraySwitch->turnOn();
134  SprayTimer = millis();
135  }
136  if (millis() - SprayTimer >= Duration * 1000)
137  {
138  LastSprayPressure = Aeroponics_Tank::FeedbackPressureSensor->getPressure();
139  updateState(AeroTankStates::STOPSPRAY);
140  BlockOverWritingState = true;
141  }
142  break;
143  case AeroTankStates::STOPSPRAY: //< Closes the spray solenoid (Takes around ~0.5sec, check solenoid specification and adjust SpraySolenoidClosingDelay in Settings.h)
144  if (State != NewState)
145  {
146  SpraySwitch->turnOff();
147  SprayTimer = millis();
148  }
149  if (millis() - SprayTimer >= SpraySolenoidClosingDelay)
150  {
151  SprayTimer = millis();
152  updateState(AeroTankStates::RELEASE);
153  BlockOverWritingState = true;
154  }
155  break;
156  case AeroTankStates::RELEASE: //< Release remaining pressure from the misting loop AND pump output side
157  if (State != NewState)
158  {
159  Pump->startBlowOff();
160  }
161  if (Pump->getState() != PressurePumpStates::BLOWOFF && Pump->getState() != PressurePumpStates::BYPASSCLOSE)
162  {
163  updateState(AeroTankStates::IDLE);
164  BlockOverWritingState = true;
165  }
166  break;
167  case AeroTankStates::REFILL:
168  if (State != NewState)
169  {
170  Pump->startPump(true);
171  }
172  if (FeedbackPressureSensor->readPressure(false) >= MaxPressure)
173  {
174  updateState(AeroTankStates::RELEASE);
175  BlockOverWritingState = true;
176  }
177  if (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED)
178  {
179  // logToSerials(F("Recharge failed"), true, 3);
180  updateState(AeroTankStates::IDLE);
181  BlockOverWritingState = true;
182  }
183  break;
184  case AeroTankStates::DRAIN:
185  if (State != NewState)
186  {
187  Pump->turnBypassOn(true);
188  SpraySwitch->turnOn();
189  }
190  if (FeedbackPressureSensor->readPressure(false) <= 0.1 || (Pump->getState() == PressurePumpStates::IDLE || Pump->getState() == PressurePumpStates::DISABLED))
191  {
192  if (Pump->getState() == PressurePumpStates::BYPASSOPEN)
193  Pump->turnBypassOff();
194  SpraySwitch->turnOff();
195  if (SprayEnabled)
196  {
197  updateState(AeroTankStates::IDLE);
198  }
199  else
200  {
201  updateState(AeroTankStates::DISABLED);
202  }
203  BlockOverWritingState = true;
204  }
205  break;
206  case AeroTankStates::MIX:
207  if (State != NewState)
208  {
209  Pump->startMixing();
210  }
211  if (Pump->getState() != PressurePumpStates::MIX)
212  {
213  updateState(AeroTankStates::IDLE);
214  BlockOverWritingState = true;
215  }
216  break;
217  }
218 
219  if (State != NewState && !BlockOverWritingState)
220  {
221  State = NewState;
222  }
