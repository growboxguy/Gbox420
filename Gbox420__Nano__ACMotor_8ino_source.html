216 void zeroCrossingInterrupt() //called every 50Hz AC : 10ms, 60Hz AC : 8.3335ms
217 { //AC signal crossed zero: start the delay before turning on the TRIAC
218  if (ACMotorMod1->Motor1->MotorState)
219  {
220  TCCR1B = *ACMotorMod1->Motor1->Prescale; // prescale the
221  TCNT1 = 0; // reset timer - count from zero
222  OCR1A = ACMotorMod1->Motor1->PID_TriacDelay; // set the compare register: triggers TIMER1_COMPA_vect when the tick counter reaches the delay calculated by the PID controller
223  }
224 }
225 
226 ISR(TIMER1_COMPA_vect) //called with a delay after each zeroCrossingInterrupt
227 { // comparator match: TRIAC delay reached after a zero crossing
228  if (ACMotorMod1->Motor1->MotorState)
229  {
230  digitalWrite(*(ACMotorMod1->Motor1->TriacPin), HIGH); // turn on TRIAC gate
231  TCNT1 = 65536 - *(ACMotorMod1->Motor1->TriacActiveTicks); // trigger pulse width
232  }
233 }
234 
235 ISR(TIMER1_OVF_vect)
236 { // timer1 overflow
237  digitalWrite(*(ACMotorMod1->Motor1->TriacPin), LOW); // turn off TRIAC gate
238  TCCR1B = 0x00; // disable timer stops unintended triggers
239 }
