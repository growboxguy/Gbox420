267 {
268  //Wireless.flush_tx(); ///< Dump all previously cached but unsent ACK messages from the TX FIFO buffer (Max 3 are saved)
269  NextSequenceID = NewSequenceID; // update the next Message that will be copied to the buffer
270

272  {
273  case AeroMessages::AeroModuleResponse1:
