195 static bool SyncInProgress = false;
196 time_t getNtpTime()
197 {
198  time_t NTPResponse = 0;
199  if (!SyncInProgress)
200  {
201  SyncInProgress = true;
202  uint32_t LastRefresh = millis();
203  logToSerials(F("Waiting for NTP time (15sec timeout)..."), false, 0);
204  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
205  {
206  NTPResponse = ESPCmd.GetTime();
207  delay(500);
208  logToSerials(F("."), false, 0);
209  wdt_reset();
210  }
211  SyncInProgress = false;
212  if (NTPResponse == 0)
213  {
214  logToSerials(F("NTP time sync failed"), true, 1);
216  }
217  else
218  logToSerials(F("time synchronized"), true, 1);
219  }
220  return NTPResponse;
221 }
