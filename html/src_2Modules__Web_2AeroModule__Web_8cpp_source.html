291 AeroMessages AeroModule_Web::sendCommand(void *CommandToSend)
292 {
293  AeroMessages SequenceIDToSend = ((AeroCommonTemplate *)CommandToSend)->SequenceID;
294  AeroMessages ReceivedSequenceID = NULL;
295  if (*Debug)
296  {
297  logToSerials(F("Sending SequenceID:"), false, 3);
298  logToSerials(SequenceIDToSend, false, 1);
299  logToSerials(F("-"), false, 1);
300  logToSerials(toText_aeroSequenceID(SequenceIDToSend), false, 1);
301  logToSerials(F("and waiting for Acknowledgment..."), true, 1);
302  }
303  Parent->Wireless->openWritingPipe(WirelessChannel);
304  Parent->Wireless->flush_rx();
305  if (Parent->Wireless->write(CommandToSend, WirelessPayloadSize))
306  {
307  delay(50);
308  if (Parent->Wireless->isAckPayloadAvailable())
309  {
310  OnlineStatus = true;
311  Parent->Wireless->read(ReceivedResponse, WirelessPayloadSize);
312  ReceivedSequenceID = ((AeroCommonTemplate *)ReceivedResponse)->SequenceID;
313  if (*Debug)
314  {
315  logToSerials(F("Response SequenceID:"), false, 4);
316  logToSerials(ReceivedSequenceID, false, 1);
317  logToSerials(F("-"), false, 1);
318  logToSerials(toText_aeroSequenceID(ReceivedSequenceID), true, 1);
319  }
320 
321  switch (ReceivedSequenceID)
322  {
323  case AeroMessages::AeroModuleResponse1:
324  memcpy(&AeroModuleResponse1Received, ReceivedResponse, sizeof(struct AeroModuleResponse));
325  if (*Debug)
326  {
327  logToSerials(F("Module1:"), false, 4);
328  logToSerials(AeroModuleResponse1Received.Status, true, 1);
329  }
330  break;
331  case AeroMessages::AeroResponse1:
332  memcpy(&AeroResponse1Received, ReceivedResponse, sizeof(struct AeroResponse_P1));
333  if (AeroCommand1ToSend.SprayEnabled || AeroCommand1ToSend.SprayDisabled || AeroCommand1ToSend.SprayNow || AeroCommand1ToSend.SprayOff)
334  {
335  SyncRequested = true;
336  AeroCommand1ToSend.SprayEnabled = false;
337  AeroCommand1ToSend.SprayDisabled = false;
338  AeroCommand1ToSend.SprayNow = false;
339  AeroCommand1ToSend.SprayOff = false;
340  }
341  if (*Debug)
342  {
343  logToSerials(F("Aero1:"), false, 4);
344  logToSerials(AeroResponse1Received.PressureTankPresent, false, 1);
345  logToSerials(F(","), false, 1);
346  logToSerials(AeroResponse1Received.SprayEnabled, false, 1);
347  logToSerials(F(","), false, 1);
348  logToSerials(AeroResponse1Received.State, false, 1);
349  logToSerials(F(","), false, 1);
350  logToSerials(AeroResponse1Received.Pressure, false, 1);
351  logToSerials(F(","), false, 1);
352  logToSerials(AeroResponse1Received.LastSprayPressure, false, 1);
353  logToSerials(F(","), false, 1);
354  logToSerials(AeroResponse1Received.Weight, true, 1);
355  }
356  break;
357  case AeroMessages::AeroResponse2:
358  memcpy(&AeroResponse2Received, ReceivedResponse, sizeof(struct AeroResponse_P2));
359  if (AeroCommand2ToSend.PumpOn || AeroCommand2ToSend.PumpOff || AeroCommand2ToSend.PumpDisable || AeroCommand2ToSend.MixReservoir || AeroCommand2ToSend.RefillPressureTank || AeroCommand2ToSend.TareWeight)
360  {
361  SyncRequested = true;
362  AeroCommand2ToSend.PumpOn = false;
363  AeroCommand2ToSend.PumpOff = false;
364  AeroCommand2ToSend.PumpDisable = false;
365  AeroCommand2ToSend.MixReservoir = false;
366  AeroCommand2ToSend.RefillPressureTank = false;
367  AeroCommand2ToSend.TareWeight = false;
368  }
369  break;
370  case AeroMessages::AeroReset:
371  if (*Debug)
372  {
373  logToSerials(F("Last message received"), true, 4);
374  }
375  break;
376  default:
377  logToSerials(F("SequenceID not known, ignoring package"), true, 4);
378  break;
379  }
380  LastResponseReceived = millis();
381  }
382  else
383  {
384  if (*Debug)
385  logToSerials(F("Acknowledgement received without any data"), true, 4);
386  OnlineStatus = false;
387  }
388  }
389  else
390  {
391  if (*Debug)
392  logToSerials(F("No response"), true, 3);
393  OnlineStatus = false;
394  }
395  return ReceivedSequenceID;
396 }
