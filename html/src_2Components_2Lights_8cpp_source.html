147 void Lights::checkTimer()
148 {
149  if (*TimerEnabled)
150  {
151  time_t Now = now(); // Get the current time from TimeLib
152  int CombinedOnTime = *OnHour * 100 + *OnMinute; // convert time to number, Example: 8:10=810, 20:10=2010
153  int CombinedOffTime = *OffHour * 100 + *OffMinute;
154  int CombinedCurrentTime = hour(Now) * 100 + minute(Now);
155  if (CombinedOnTime <= CombinedOffTime) // no midnight turnover, Example: On 8:10, Off: 20:10
156  {
157  if (CombinedOnTime <= CombinedCurrentTime && CombinedCurrentTime < CombinedOffTime)
158  { // True: Light should be on
159  if (!*Status)
160  {
161  setLightOnOff(true, false); // If status is OFF: Turn ON the lights (First bool), and do not add it to the log (Second bool)
162  if (*Debug)
163  logToSerials(F("Timer:Light ON"), true, 4);
164  }
165  }
166  else // False: Light should be off
167  {
168  if (*Status)
169  { // If status is ON
170  setLightOnOff(false, false); // Turn OFF the lights (First bool), and do not add it to the log (Second bool)
171  if (*Debug)
172  logToSerials(F("Timer:Light OFF"), true, 4);
173  }
174  }
175  }
176  else // midnight turnover, Example: On 21:20, Off: 9:20
177  {
178  if (CombinedOnTime <= CombinedCurrentTime || CombinedCurrentTime < CombinedOffTime)
179  {
180  if (!*Status)
181  {
182  setLightOnOff(true, false);
183  if (*Debug)
