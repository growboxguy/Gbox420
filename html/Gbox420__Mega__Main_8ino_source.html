200 time_t getNtpTime()
201 {
202  time_t NTPResponse = 0;
203  if (!SyncInProgress)
204  {
205  SyncInProgress = true;
206  uint32_t LastRefresh = millis();
207  logToSerials(F("Waiting for NTP time..."), false, 0);
208  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
209  {
210  NTPResponse = ESPCmd.GetTime();
211  delay(500);
212  logToSerials(F("."), false, 0);
213  wdt_reset();
214  }
215  SyncInProgress = false;
216  if (NTPResponse == 0)
217  {
218  logToSerials(F("NTP time sync failed"), true, 1);
220  }
221  else
222  logToSerials(F("time synchronized"), true, 1);
223  }
224  return NTPResponse;
225 }
