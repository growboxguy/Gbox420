219 void HempyModule::updateAckData()
220 { // so you can see that new data is being sent
221  if (*Debug)
222  {
223  logToSerials(F("Updating Acknowledgement to:"), false, 2);
224  logToSerials(toText_hempySequenceID(NextSequenceID), true, 1);
225  }
226  Wireless.flush_tx();
227 
228  switch (NextSequenceID) // based on the NextSeqenceID load the next response into the Acknowledgement buffer
229  {
230  case HempyMessages::HempyModuleResponse1:
231  Wireless.writeAckPayload(1, &HempyModuleResponse1ToSend, WirelessPayloadSize);
232  break;
233  case HempyMessages::HempyBucketResponse1:
234  Wireless.writeAckPayload(1, &HempyBucket1ResponseToSend, WirelessPayloadSize);

236  case HempyMessages::HempyBucketResponse2:
237  Wireless.writeAckPayload(1, &HempyBucket2ResponseToSend, WirelessPayloadSize);
238  break;
239  case HempyMessages::HempyReset:
240  Wireless.writeAckPayload(1, &HempyResetToSend, WirelessPayloadSize);

245  logToSerials(F("Unknown next Sequence number, Ack defaults loaded"), true, 3);
