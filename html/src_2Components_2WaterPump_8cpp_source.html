80 {

86 
87  switch (State)

89  case PRIMING:

97  break;
98  case RUNNING:
99  PumpOn = true;
100  BypassOn = false;

104  RunTime = 0;
105  logToSerials(F("Running complete, stopping..."), true, 3);
106  if (BlowOffTime != NULL && *BlowOffTime > 0)
107  {
108  updateState(BLOWOFF);
109  }
110  else
111  {
112  updateState(IDLE);
113  }
114  }
115  if (millis() - PumpTimer > ((uint32_t)*PumpTimeOut * 1000))
116  {
117  Parent->addToLog(F("ALERT: Pump timeout reached"), 3);
118  updateState(DISABLED);
119  }
120  break;
121  case BLOWOFF:
122  PumpOn = false;
123  BypassOn = true;
124  *PumpEnabled = true;
125  if (millis() - PumpTimer > ((uint32_t)*BlowOffTime * 1000))
126  {
127  logToSerials(F("Pressure released"), true, 3);
128  updateState(IDLE);
129  }
130  break;
131  case IDLE:
132  PumpOn = false;
133  BypassOn = false;
134  *PumpEnabled = true;
135  break;
136  case MIXING:
137  PumpOn = true;
138  BypassOn = true;
139  *PumpEnabled = true;
140  if ((RunTime > 0 && millis() - PumpTimer > ((uint32_t)RunTime * 1000)) || millis() - PumpTimer > ((uint32_t)*PumpTimeOut * 1000))
141  {
142  RunTime = 0;
143  logToSerials(F("Mixing finished"), true, 3);
144  updateState(IDLE);
145  }
146  break;
147  case DISABLED:
148  PumpOn = false;
149  BypassOn = false;
150  *PumpEnabled = false;
151  break;
