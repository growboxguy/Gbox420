103 /* Yield equivalent to "*portITU_SWINTR = 0x01; ( void ) *portITU_SWINTR;"
104  * where portITU_SWINTR is the location of the software interrupt register
105  * (0x000872E0). Don't rely on the assembler to select a register, so instead
106  * save and restore clobbered registers manually. */
107 #define portYIELD() \
108  __asm volatile \
109  ( \
110  "PUSH.L R10 \n" \
111  "MOV.L #0x872E0, R10 \n" \
112  "MOV.B #0x1, [R10] \n" \
113  "CMP [R10].UB, R10 \n" \
114  "POP R10 \n" \
115  ::: "cc" \
116  )

130 /* These macros should not be called directly, but through the
131  * taskENTER_CRITICAL() and taskEXIT_CRITICAL() macros. An extra check is
132  * performed if configASSERT() is defined to ensure an assertion handler does not
133  * inadvertently attempt to lower the IPL when the call to assert was triggered
134  * because the IPL value was found to be above configMAX_SYSCALL_INTERRUPT_PRIORITY
135  * when an ISR safe FreeRTOS API function was executed. ISR safe FreeRTOS API
136  * functions are those that end in FromISR. FreeRTOS maintains a separate
137  * interrupt API to ensure API function and interrupt entry is as fast and as
138  * simple as possible. */
139 #define portENABLE_INTERRUPTS() __asm volatile ( "MVTIPL #0" )
140 #ifdef configASSERT
141  #define portASSERT_IF_INTERRUPT_PRIORITY_INVALID() configASSERT( ( ulPortGetIPL() <= configMAX_SYSCALL_INTERRUPT_PRIORITY ) )
142  #define portDISABLE_INTERRUPTS() if( ulPortGetIPL() < configMAX_SYSCALL_INTERRUPT_PRIORITY ) __asm volatile( "MVTIPL %0" ::"i" ( configMAX_SYSCALL_INTERRUPT_PRIORITY ) )
