36 static inline void pwm_calcDivTop(pwm_config *c, int frequency, int sysClock)
37 {
38  uint count = 2000000000 / frequency;
39  uint div = count / 60000; // to be lower than 65535*15/16 (rounding error)
40  if (div < 16)
41  div = 16;
42  c->div = div;
43  c->top = count / div;
44 }
