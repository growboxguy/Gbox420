158 void ACMotorModule::updateAckData(ACMotorMessages NewSequenceID)
159 {
160  //Wireless.flush_tx(); ///< Dump all previously cached but unsent ACK messages from the TX FIFO buffer (Max 3 are saved)
161  NextSequenceID = NewSequenceID; // update the next Message that will be copied to the buffer
162 
163  switch (NextSequenceID) // based on the NextSeqenceID load the next response into the Acknowledgement buffer
164  {
165  case ACMotorMessages::ACMotorModuleResponse1:
166  Wireless.writeAckPayload(1, &ACMotorModuleResponse1ToSend, WirelessPayloadSize);
167  break;
168  case ACMotorMessages::ACMotorResponse1:
169  Wireless.writeAckPayload(1, &ACMotor1ResponseToSend, WirelessPayloadSize);
170  break;
171  case ACMotorMessages::ACMotorReset:
172  Wireless.writeAckPayload(1, &ACMotorResetToSend, WirelessPayloadSize);
173  break;
174  default:
175  Wireless.writeAckPayload(1, &ACMotorModuleResponse1ToSend, WirelessPayloadSize); // load the first Response into the buffer
