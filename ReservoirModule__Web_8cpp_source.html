126 ReservoirMessages ReservoirModule_Web::sendCommand(void *CommandToSend)
127 {
128  ReservoirMessages SequenceIDToSend = ((ReservoirCommonTemplate *)CommandToSend)->SequenceID;
129  ReservoirMessages ReceivedSequenceID = NULL;
130  if (Parent->SerialReportWireless)
131  {

134  }
135  Parent->Wireless->openWritingPipe(WirelessChannel);
136  Parent->Wireless->flush_rx();
137  if (Parent->Wireless->write(CommandToSend, WirelessPayloadSize))
138  {
139  delay(50);

141  {
142  OnlineStatus = true;
143  Parent->Wireless->read(ReceivedResponse, WirelessPayloadSize);
144  ReceivedSequenceID = ((ReservoirCommonTemplate *)ReceivedResponse)->SequenceID;
145  if (Parent->SerialReportWireless)
146  {
147  logToSerials(F("; Response:"), false, 1);
148  logToSerials(toText_reservoirSequenceID(ReceivedSequenceID), false, 1);
149  logToSerials(F("; Data:"), false, 1);
150  }
151 
152  switch (ReceivedSequenceID)
153  {

155  memcpy(&ReservoirModuleResponse1Received, ReceivedResponse, sizeof(struct ReservoirModuleResponse));
156  if (Parent->SerialReportWireless)
157  {
158  logToSerials(ReservoirModuleResponse1Received.Status, true, 1);
159  }
160  break;
161  case ReservoirMessages::ReservoirResponse1:
162  memcpy(&ReservoirResponse1Received, ReceivedResponse, sizeof(struct ReservoirResponse));
163  if (Parent->SerialReportWireless)
164  {
165  logToSerials(ReservoirResponse1Received.ConfirmTareWeight, false, 1);
166  logToSerials(ReservoirResponse1Received.PH, false, 1);

172  logToSerials(ReservoirResponse1Received.Humidity, true, 1);
173  }
174  if (ReservoirCommand1ToSend.TareWeight || ReservoirCommand1ToSend.TareWeightWR)
175  {
176  SyncRequested = true;
177  }
178  if (ReservoirResponse1Received.ConfirmTareWeight)
179  ReservoirCommand1ToSend.TareWeight = false; // Turn off the Flag once the Receiver confirms processing it
180  if (ReservoirResponse1Received.ConfirmTareWeightWR)

193  }
194  break;
195  }

197  }
198  else

202  }
203  }
204  else

210  OnlineStatus = false;
211  }
212  }
