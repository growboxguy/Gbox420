238  static char command[MaxShotTextLength];

260 time_t getNtpTime()
261 {
262  time_t NTPResponse = 0;
263  if (!SyncInProgress)
264  { // block calling the sync again inside an interrupt while already waiting for a sync to finish
265  SyncInProgress = true;
266  uint32_t LastRefresh = millis();
267  logToSerials(F("Waiting for NTP time"), false, 0);
268  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
269  {
270  NTPResponse = ESPCmd.GetTime();
271  delay(500);
272  logToSerials(F(""), false, 0);
273  wdt_reset();
274  }
275  SyncInProgress = false;
276  if (NTPResponse == 0)
277  {
278  logToSerials(F("NTP time sync failed"), true, 1);
279  // sendEmailAlert(F("NTP%20time%20sync%20failed"));
280  }
281  else
282  logToSerials(F("time synchronized"), true, 3);
283  }
284  return NTPResponse;
285 }
