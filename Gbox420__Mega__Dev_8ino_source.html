257 time_t getNtpTime()
258 {
259  time_t NTPResponse = 0;
260  if (!SyncInProgress)
261  { // block calling the sync again inside an interrupt while already waiting for a sync to finish
262  SyncInProgress = true;
263  uint32_t LastRefresh = millis();
264  logToSerials(F("Waiting for NTP time"), false, 0);
265  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
266  {
267  NTPResponse = ESPCmd.GetTime();
268  delay(1000);
269  logToSerials(F("."), false, 0);
270  wdt_reset();
271  }
272  SyncInProgress = false;
273  if (NTPResponse == 0)
274  {
275  logToSerials(F("sync failed"), true, 3);
276  }
277  else
278  logToSerials(F("synchronized"), true, 3);
279  }
280  return NTPResponse;
281 }
